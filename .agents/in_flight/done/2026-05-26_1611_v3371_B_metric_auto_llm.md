### 2.1 Step 1→2 transition: LLM önce, fallback sonra

`_loadMetrics()` fonksiyonunu refactor:

```js
async function _loadMetrics() {
    const panel = document.getElementById('dswStep2');
    if (!panel) return;
    if (!_state.sourceId) {
        panel.innerHTML = '<p class="dsw-hint" role="status">' + _escape(_t('wizard.hint.select_source_first')) + '</p>';
        return;
    }
    _setBusy(panel, true);

    // YENİ — LLM auto-suggest spinner mesajı
    panel.innerHTML = '<p class="dsw-hint" role="status">🤖 METIS metrikleri analiz ediyor...</p>';

    // 1) LLM auto-suggest dene (table_id + columns + FK + amaç)
    let llmSuggestions = null;
    let llmError = null;
    const llmAbort = new AbortController();
    _state._metricLlmAbort = llmAbort;  // step out ederse cancel

    if (_state.selectedTableId) {
        try {
            const llmData = await _autoSuggestMetrics({
                source_id: _state.sourceId,
                table_id: _state.selectedTableId,
                table_label: _state.selectedTableLabel,
                joined_tables: _state.selectedTables || [],
                signal: llmAbort.signal,
            });
            if (llmData && Array.isArray(llmData.suggestions) && llmData.suggestions.length) {
                llmSuggestions = llmData.suggestions;
            }
        } catch (e) {
            if (e.name === 'AbortError') return;  // step out
            llmError = e;
            console.warn('[db_smart_wizard] auto metric-suggest failed:', e);
        }
    }

    // 2) Static library çek (her zaman — fallback + ek)
    let staticItems = [];
    let staticFallback = false;
    try {
        let url = API_BASE + '/metrics?source_id=' + _state.sourceId;
        if (_state.selectedTableId) url += '&table_id=' + _state.selectedTableId;
        const data = await _fetchJson(url);
        staticItems = data.items || [];
        staticFallback = !!(data && data.fallback);
    } catch (e) {
        console.warn('[db_smart_wizard] static metrics failed:', e);
    }

    // 3) Merge: LLM önce ("✨ LLM Önerisi"), sonra statik kategoriler
    _state._metricsIndex = {};
    _state._metricCategories = {};

    if (llmSuggestions) {
        const llmCat = '✨ LLM Önerisi';
        _state._metricCategories[llmCat] = llmSuggestions.map((s, i) => {
            const key = 'llm_' + (s.metric_name || 'metric_' + i).replace(/\W+/g, '_').toLowerCase();
            const item = {
                metric_key: key,
                metric_name: s.metric_name,
                formula: s.formula,
                agg: s.agg,
                rationale: s.rationale,
                confidence: s.confidence,
                category: llmCat,
                source: 'llm',
            };
            _state._metricsIndex[key] = item;
            return item;
        });
    }

    staticItems.forEach(m => {
        if (!m || !m.metric_key || _state._metricsIndex[m.metric_key]) return;
        _state._metricsIndex[m.metric_key] = m;
        const cat = m.category || (m.applicable_when && m.applicable_when.category) || 'Diğer';
        (_state._metricCategories[cat] = _state._metricCategories[cat] || []).push(m);
    });

    // 4) Stale selectedMetrics temizle (mevcut davranış korunur)
    if (_state.selectedMetrics && typeof _state.selectedMetrics.forEach === 'function') {
        const toDrop = [];
        _state.selectedMetrics.forEach(k => {
            if (!_state._metricsIndex[k]) toDrop.push(k);
        });
        toDrop.forEach(k => _state.selectedMetrics.delete(k));
    }

    const totalCount = Object.values(_state._metricsIndex).length;
    if (!totalCount) {
        panel.innerHTML = '<p class="dsw-hint">' + _escape(_t('wizard.empty.metrics')) + '</p>';
        _setBusy(panel, false);
        return;
    }

    // 5) Fallback notu: SADECE LLM başarısız VE static fallback durumunda göster
    const showFallbackNote = (!llmSuggestions) && staticFallback;
    _renderStep2(Object.values(_state._metricsIndex), showFallbackNote);
    _setBusy(panel, false);
}
```

### 2.2 Yeni helper `_autoSuggestMetrics`

`_loadMetrics`'in hemen altına (~654 satırından sonra):

```js
async function _autoSuggestMetrics(payload) {
    // table metadata + FK ipuçlarını backend kendisi resolve eder
    const body = {
        source_id: payload.source_id,
        table_id: payload.table_id,
        table_label: payload.table_label || '',
        joined_tables: payload.joined_tables || [],
        user_intent: _state.user_intent || '',
    };
    return await _fetchJson(API_BASE + '/llm/metric-suggest', {
        method: 'POST',
        body: JSON.stringify(body),
        signal: payload.signal,
    });
}
```
