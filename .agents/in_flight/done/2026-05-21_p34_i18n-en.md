```js
(function () {
    'use strict';
    const STORAGE_KEY = 'vyra.lang';
    const FALLBACK = 'tr';
    const SUPPORTED = ['tr', 'en'];
    const _bundles = {};  // {lang: {key: str}}
    let _currentLang = null;

    function detectLang() {
        // Priority: URL ?lang= > localStorage > navigator.language > FALLBACK
        try {
            const u = new URLSearchParams(window.location.search).get('lang');
            if (u && SUPPORTED.includes(u)) return u;
        } catch (e) {}
        try {
            const s = localStorage.getItem(STORAGE_KEY);
            if (s && SUPPORTED.includes(s)) return s;
        } catch (e) {}
        try {
            const n = (navigator.language || 'tr').slice(0, 2).toLowerCase();
            if (SUPPORTED.includes(n)) return n;
        } catch (e) {}
        return FALLBACK;
    }

    async function loadBundle(lang) {
        if (_bundles[lang]) return _bundles[lang];
        try {
            const res = await fetch(`/static/js/i18n/aki_kesif_${lang}.json`,
                                    {cache: 'no-cache'});
            if (!res.ok) throw new Error(`HTTP ${res.status}`);
            _bundles[lang] = await res.json();
            return _bundles[lang];
        } catch (e) {
            console.warn('[i18n] bundle load failed', lang, e);
            if (lang !== FALLBACK) return loadBundle(FALLBACK);
            return {};
        }
    }

    async function init() {
        _currentLang = detectLang();
        await loadBundle(_currentLang);
        // Also update <html lang="…"> for SR + Lighthouse
        try { document.documentElement.setAttribute('lang', _currentLang); } catch (e) {}
        return _currentLang;
    }

    function t(key, params) {
        const bundle = _bundles[_currentLang] || {};
        let raw = bundle[key];
        if (raw === undefined && _currentLang !== FALLBACK) {
            raw = (_bundles[FALLBACK] || {})[key];
        }
        if (raw === undefined) return key;  // key passthrough for debug
        if (params && typeof params === 'object') {
            return raw.replace(/\{(\w+)\}/g, (m, k) => (k in params ? String(params[k]) : m));
        }
        return raw;
    }

    function applyTranslations(rootEl) {
        if (!rootEl) return;
        rootEl.querySelectorAll('[data-i18n]').forEach(el => {
            const key = el.getAttribute('data-i18n');
            const params = el.getAttribute('data-i18n-params');
            let p = null;
            if (params) { try { p = JSON.parse(params); } catch (e) {} }
            el.textContent = t(key, p);
        });
        rootEl.querySelectorAll('[data-i18n-attr]').forEach(el => {
            // data-i18n-attr="title:tooltip.help;aria-label:button.save"
            const spec = el.getAttribute('data-i18n-attr');
            spec.split(';').forEach(pair => {
                const [attr, key] = pair.split(':').map(s => s.trim());
                if (attr && key) el.setAttribute(attr, t(key));
            });
        });
    }

    function setLang(lang) {
        if (!SUPPORTED.includes(lang)) return Promise.resolve(_currentLang);
        try { localStorage.setItem(STORAGE_KEY, lang); } catch (e) {}
        return loadBundle(lang).then(() => {
            _currentLang = lang;
            try { document.documentElement.setAttribute('lang', lang); } catch (e) {}
            return lang;
        });
    }

    function getLang() { return _currentLang; }

    window.VyraI18n = { init, t, applyTranslations, setLang, getLang, SUPPORTED };
})();
```

```python
"""Server-side i18n messages for DB Smart Wizard (FAZ 5 P34).

Used by db_smart_api error responses + state_machine event payloads when
Accept-Language is parsed and EN is preferred.
"""

from typing import Literal

Lang = Literal["tr", "en"]
DEFAULT_LANG: Lang = "tr"

_MESSAGES: dict[str, dict[Lang, str]] = {
    "ELIGIBILITY_EMPTY": {
        "tr": "Bu tabloya uygun metrik bulunamadı.",
        "en": "No eligible metrics found for this table.",
    },
    "SQL_GUARD_DENY": {
        "tr": "Sorgu güvenlik kontrolüne takıldı.",
        "en": "Query blocked by safety guard.",
    },
    "METRIC_NOT_FOUND": {
        "tr": "Metrik bulunamadı.",
        "en": "Metric not found.",
    },
    "SESSION_EXPIRED": {
        "tr": "Oturum süresi doldu.",
        "en": "Session expired.",
    },
    "SESSION_CONFLICT": {
        "tr": "Oturum başka bir cihazda güncellendi.",
        "en": "Session updated by another device.",
    },
    "AST_PATCH_INVALID": {
        "tr": "Geçersiz AST işlemi.",
        "en": "Invalid AST operation.",
    },
    "RATE_LIMITED": {
        "tr": "Çok fazla istek. Lütfen biraz bekleyin.",
        "en": "Too many requests. Please wait a moment.",
    },
    # ... ~15-20 keys total covering db_smart error surface
}


def get_message(key: str, lang: Lang = DEFAULT_LANG, **params) -> str:
    """Return localized message; fallback to TR; fallback to key on miss."""
    if key not in _MESSAGES:
        return key
    bundle = _MESSAGES[key]
    msg = bundle.get(lang) or bundle.get(DEFAULT_LANG) or key
    if params:
        try:
            return msg.format(**params)
        except (KeyError, IndexError):
            return msg
    return msg


def parse_accept_language(header: str | None) -> Lang:
    """Parse Accept-Language header; return 'en' if user prefers EN, else 'tr'.
    Simple implementation: check first language tag startswith 'en'."""
    if not header:
        return DEFAULT_LANG
    primary = header.split(",")[0].strip().split(";")[0].strip().lower()
    if primary.startswith("en"):
        return "en"
    return "tr"
```
